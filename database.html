C program to create and implement database

#include <stdio.h>
# define MAX_SIZE 10
struct comp_data
{
char comp_name[30];
int value;
float cost;

}s[MAX_SIZE];
void display(struct comp_data arr[],int n)
{
int i;
for (i=0;i {
printf("%s\t",arr[i].comp_name);
printf("%d\t",arr[i].value);
printf("%f\t",arr[i].cost);
printf("\n");
}
}
void create(struct comp_data arr[],int n)
{
int i;
for (i=0;i {
printf("\nenter %d name : ",i+1);

scanf("%s",&arr[i].comp_name);
printf("\nenter %d value : ",i+1);
scanf("%d",&arr[i].value);
printf("\nenter %d cost : ",i+1);
scanf("%f",&arr[i].cost);
}
}
int append(struct comp_data arr[],int n)
{
printf("Enter data to add\n");
printf("\nenter %d component name: ",n+1);
scanf("%s",&arr[n].comp_name);
printf("\nenter %d component value: ",n+1);
scanf("%d",&arr[n].value);
printf("\nenter %d component cost : ",n+1);
scanf("%f",&arr[n].cost);
n+=1;
return n;
}
void modify(struct comp_data arr[],int n)
{
int location;
printf("Enter location at which you want to modify element\n");
scanf("%d",&location);
if (location>n)
{
printf("Modification not possible\n");
}
else
{
printf("Enter data to add\n");
printf("\nenter %d name : ",location);
scanf("%s",&arr[location-1].comp_name);
printf("\nenter %d value : ",location);
scanf("%d",&arr[location-1].value);
printf("\nenter %d cost : ",location);
scanf("%f",&arr[location-1].cost);
printf("modified data is\n");
display(arr, n);
}
}
void search_data(struct comp_data arr[],int n)
{
int val;
int i;
printf("Enter value to search\n");
scanf("%d",&val);
for (i=0;i {
if (arr[i].value==val)
{
printf("data found at location %d\n",i+1);
printf("component name : %s\n",arr[i].comp_name);
printf("component value : %d\n",arr[i].value);
printf("component cost : %f\n",arr[i].cost);
break;
}
}
if (i==n)
{
printf("NOT FOUND\n");
}
}
void sort_data(struct comp_data arr[],int n)
{
struct comp_data temp;
int i , j;
for (i=0;i {
for (j=0; j < (n-i-1);j++)
{
if(arr[j].value>arr[j+1].value)
{
temp=arr[j];
arr[j]=arr[j+1];
arr[j+1]=temp;
}
}
}
printf("SORTED DATABASE ACCORDING TO VALUE IS\n");
display(arr, n);
}
void main()
{
int size,choice,choice1;
printf("Enter no of components : ");
scanf("%d",&size);
create(s,size);
display(s,size);
do
{

printf("\nEnter operation you want to perform\n1.APPEND\n2.MODIFY\n3.SEARCH\n4.SORT DATA\n");
scanf("%d",&choice);
switch (choice)
{
case 1:
size=append(s,size);
break;
case 2:
modify(s,size);
break;
case 3:
search_data(s,size);
break;
case 4:
sort_data(s,size);
break;
default:
printf("INVALID CHOICE\n");
}
printf("Press1 to continue\n");
scanf("%d",&choice1);


}while(choice1==1);
}
