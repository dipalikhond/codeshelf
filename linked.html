C program to perform operations on linked list

#include <stdio.h>
#include <stdlib.h>
struct node{
int data;
struct node *next;
};
void display(struct node *ptr)
{
while(ptr!=NULL)
{
printf("%d\n",ptr->data);
ptr=ptr->next;
}

}
struct node* insert_at_beg(struct node *head,int val)
{
printf("INSERT FUNC IS CALLED\n");
struct node *p;
p=(struct node *) malloc(sizeof(struct node));
p->data=val;
p->next=head;
return p;

}
void insert_at_end(struct node *head,int val)
{
printf("INSERT FUNC IS CALLED\n");
struct node *p;
p=(struct node *) malloc(sizeof(struct node));
p->data=val;
while (head->next!=NULL)
{
head=head->next;
}
head->next=p;
p->next=NULL;


}
struct node* insert_at_index(struct node *head,int val,int index)
{
printf("INSERT FUNC IS CALLED\n");
struct node *p;
p=(struct node *) malloc(sizeof(struct node));
p->data=val;
int i=0;
while (i!=(index-1))
{
head=head->next;
i++;
}
p->next=head->next;
head->next=p;


}
struct node* del_at_beg(struct node* ptr)
{
printf("DELETE FUNC IS CALLED\n");
struct node* p=(struct node *) malloc(sizeof(struct node));
p=ptr;
ptr=ptr->next;
free(p);
return ptr;
}
void del_at_end(struct node* ptr)
{
printf("DELETE FUNC IS CALLED\n");
struct node* p=(struct node *) malloc(sizeof(struct node));
struct node* q=(struct node *) malloc(sizeof(struct node));
p=ptr;
q=p->next;
while (q->next!=NULL)
{
p=p->next;
q=q->next;
}
p->next=NULL;
free(q);


}

int size(struct node* ptr)
{
int i;

while (ptr!=NULL)
{

void reverse(struct node* head)
{
if (head == NULL)
{
return;
}
reverse(head->next);
printf("%d ", head->data);
printf("\n");
}
struct node* reverseList(struct node* head)
{ struct node *prevNode, *curNode;

if(head != NULL)
{
prevNode = head;
curNode = head->next;
head = head->next;
prevNode->next = NULL; // Make first node as last node
while(head != NULL)
{
head = head->next;
curNode->next = prevNode;

prevNode = curNode;
curNode = head;
}

head = prevNode; // Make last node as head

printf("SUCCESSFULLY REVERSED LIST\n");
return head;
}
}
