#include <stdio.h>
int length(char str[30])
{
int i; int len=0;
for (i=0;str[i]!='\0';i++)
{
len++;
}
return len;
}

void is_palindrome(char str[30])
{
char arr[30];
int i;
for(i=0;i {
arr[i]=str[i];
}
for(i=0;i {
if(str[i]!=arr[length(arr)-i-1])
{
printf("strings are not palindrome\n");
break;
}
}
if(i==length(str))
{
printf("palindrome\n");
}
}

void substring(char str[30])
{
char search[10];
int i, j, flag;
printf("Enter search substring:\n");
getchar();
gets(search);
for (i = 0; i <= length(str) - length(search); i++)
{
for (j = i; j < i + length(search); j++)
{
flag = 1;
if (str[j] != search[j - i])
{
flag = 0;
break;
}
}
if (flag == 1)
break;
}
if (flag == 1)
printf("SUBSTRING FOUND!");
else
printf("SUBSTRING NOT FOUND!");
}
void reverse(char str[30])
{
char arr[30];
int i;
for (i=0;i {
arr[i]=str[length(str)-i-1];
}
printf("REVERSE STRING IS\n");
printf("%s",arr);
}
