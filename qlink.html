C program to perform operations on Queue (using linked list)

#include <stdio.h>
#include <stdlib.h>
struct node
{
int data;
struct node *next;
};
struct node *front;
struct node *rear;
void insert()
{
struct node *ptr;
int item;

ptr = (struct node *) malloc (sizeof(struct node));
if(ptr == NULL)
{
printf("\nOVERFLOW\n");
return;
}
else
{
printf("\nEnter value?\n");
scanf("%d",&item);
ptr -> data = item;
if(front == NULL)
{
front = ptr;
rear = ptr;
front -> next = NULL;
rear -> next = NULL;
}
else
{
rear -> next = ptr;
rear = ptr;
rear->next = NULL;
}
}
}
void deleted ()
{
struct node *ptr;
if(front == NULL)
{
printf("\nUNDERFLOW\n");
return;
}
else
{
ptr = front;
front = front -> next;
free(ptr);
}
}
void display()
{
struct node *ptr;
ptr = front;
if(front == NULL)
{
printf("\nEmpty queue\n");
}
else
{ printf("\nprinting values .....\n");
while(ptr != NULL)
{
printf("%d\t",ptr -> data);
ptr = ptr -> next;
}
}
}
void main ()
{
printf("ROLL NO : 22318\n");
int choice;
int choice1;
do
{
printf("Enter choice :\n 1. ADD ELEMENT \n 2. DELETE ELEMENT \n 3. DISPLAY\n");
scanf("%d",&choice);
switch (choice)
{
case 1:
insert();
break;
case 2:
deleted();
break;
case 3:
display();
break;
default:
printf("INVALID CHOICE......\n");
break;

}
printf("\nPRESS 1 TO CONTINUE.........\n");
scanf("%d",&choice1);

}while (choice1==1);
}
