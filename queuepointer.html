Queue with pointers

#include <stdio.h>
# define MAXIMUM 10
struct queue{
int q[MAXIMUM];
int front , rear;

}*que;
void add()
{
if (que->rear==(MAXIMUM-1))
{
printf("QUEUE IS FULL..........\n");
}
else
{
que->rear++;
printf("Enter element to enter in a queue : ");
scanf("%d",&que->q[que->rear]);

}


}
void display()
{
if (que->front==-1)
{
printf("\nQUEUE IS EMPTY...........\n");
}
else
{
int i;
printf("\nQUEUE\n");
for(i=que->front;i<=que->rear;i++)
{

printf("%d\t",que->q[i]);
}
printf("\n");
}

}
void del()
{
if(que->rear==(que->front-1))
{
printf("QUEUE IS EMPTY , NOTHING TO DELETE\n");
}
else
{
int i;

printf("\nThe number %d is successfully popped from the queue....\n",que->q[que->front]);
que->front++;
}

}
void main()
{
printf("ROLL NO : 22318\n");
que->front=0;
que->rear=-1;
int choice;
int choice1;
do
{
printf("Enter choice :\n 1. ADD ELEMENT \n 2. DELETE ELEMENT \n 3. DISPLAY\n");
scanf("%d",&choice);
switch (choice)
{
case 1:
add();
break;
case 2:
del();
break;
case 3:
display();
break;
default:
printf("INVALID CHOICE......\n");
break;

}
printf("\nPRESS 1 TO CONTINUE.........\n");
scanf("%d",&choice1);

}while (choice1==1);

}
